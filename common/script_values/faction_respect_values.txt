# TODO: Only using this version until Aeluran Respect is properly added
aeluran_respect_score_by_title = {
	value = 0
	add = {
		# Elf Blood
		if = {
			limit = {
				is_elf_blood = yes
			}
			if = {
				limit = {
					# Mid Respect, your only half elf
					highest_held_title_tier >= tier_kingdom
				}
				add = 3
			}
			else_if = {
				limit = {
					highest_held_title_tier = tier_duchy
				}
				add = 2
			}
			else_if = {
				limit = {
					highest_held_title_tier = tier_county
				}
				add = 1
			}
		}
		# Elf
		if = {
			limit = {
				is_elf = yes
			}
			if = {
				limit = {
					highest_held_title_tier = tier_empire
				}
				add = 5
			}
			else_if = {
				limit = {
					# High Respect
					highest_held_title_tier = tier_kingdom
				}
				add = 4
			}
			else_if = {
				limit = {
					# Neutral Respect
					highest_held_title_tier = tier_duchy
				}
				add = 3
			}
			else_if = {
				limit = {
					# Low Respect
					highest_held_title_tier = tier_county
				}
				add = 2
			}
			else = {
				add = 1
			}
		}
		# High/True Elf
		if = {
			limit = {
				OR = {
					is_high_elf = yes
					is_true_elf = yes
					is_fae = yes
					is_fae_radiant = yes
					is_celestial = yes
					is_seraphic_celestial = yes
					is_eldar = yes
					is_maiar = yes
					is_valar = yes
					is_aratar = yes
				}
			}
			if = {
				limit = {
					# Very High Respect
					highest_held_title_tier = tier_empire
				}
				add = 5
			}
			else_if = {
				limit = {
					# High Respect
					highest_held_title_tier = tier_kingdom
				}
				add = 4
			}
			else_if = {
				limit = {
					# Neutral Respect
					highest_held_title_tier = tier_duchy
				}
				add = 3
			}
			else = {
				add = 2
			}
		}
	}
}
